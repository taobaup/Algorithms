```
  1                        n = 1

                2        1                   n = 2
               /          \
              1            2
  
   1         3     3      2      1           n = 3
    \       /     /      / \      \
     3     2     1      1   3      2
    /     /       \                 \
   2     1         2                 3
```

```
n分别为1,2,3

dp[2]=dp[0]*dp[1]　　　(1为根的情况)

　　　+dp[1]*dp[0]　 (2为根的情况)


dp[3]=dp[0]*dp[2]　　  (1为根的情况)

　　　+dp[1]*dp[1]　 (2为根的情况)

 　　 +dp[2]*dp[0]　 (3为根的情况)
```

```
class Solution {
public:
    int numTrees(int n) {
        vector<int> dp(n + 1, 0);
        dp[0] = 1;
        dp[1] = 1;

        for(int i = 2; i <= n; ++i)
        {
            for(int j = 0; j < i; ++j)
            {
                // 不要把 * 写成 +
                dp[i] += dp[j] * dp[i - j - 1];
            }
        }

        return dp[n];
    }
};
```
