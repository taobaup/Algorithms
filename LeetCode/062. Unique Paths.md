class Solution {
public:
    int uniquePaths(int m, int n) {
        dp = vector<vector<int>>(m, vector<int>(n, 0));
        dp[0][0] = 1;
        int res = dfs(m - 1, n - 1);

        return res;
    }

private:
    vector<vector<int>> dp;

    int dfs(int x, int y)
    {
        if(x < 0 || y < 0)
            return 0;

        if((x > 0 || y > 0) && dp[x][y] == 0)
            // 不要写成 dp[x][y] = dp[x][y - 1] + dp[x - 1][y];
            dp[x][y] = dfs(x, y - 1) + dfs(x - 1, y);

        return dp[x][y];
    }
};
