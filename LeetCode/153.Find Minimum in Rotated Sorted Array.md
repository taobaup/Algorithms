class Solution {
public:
    int findMin(vector<int>& nums) {
        const int len = nums.size();
        if(len == 0)
            return 0;

        int left = 0;
        int right = len - 1;
        while(left < right)
        {
            int mid = left + (right - left) / 2;
            if (nums[mid] < nums[right])
            {
                right = mid;
            }
            else if (nums[mid] > nums[right])
            {
                left = mid + 1;
            }
        }
        
        return nums[left];
    }
};
