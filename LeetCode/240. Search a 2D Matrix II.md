class Solution {
public:
    bool searchMatrix(vector<vector<int>>& matrix, int target) {
        if(matrix.empty())
        	return false;

        const int row = matrix.size();
        const int column = matrix[0].size();

        int i = 0;
        int j = column - 1;
        while(i <= row - 1 && j >= 0)
        {
        	if(target == matrix[i][j])
        		return true;
        	else if(target < matrix[i][j])
        		--j;
        	else
        		++i;
        }

        return false;
    }
};
