INT_MIN: -2147483648
INT_MAX: +2147483647

```
class Solution {
public:
    int myAtoi(string str) {
        if (str.size() == 0) 
            return 0;

        int sign = 1;
        int res = 0;
        int i = 0;
        int n = str.size();

        while (i < n && str[i] == ' ') 
            ++i;

        if (str[i] == '+' || str[i] == '-') 
        {
            sign = (str[i++] == '+') ? 1 : -1;
        }

        while (i < n && str[i] >= '0' && str[i] <= '9') 
        {
            if (res > INT_MAX / 10 || (res == INT_MAX / 10 && str[i] - '0' > 7)) 
            {
                return (sign == 1) ? INT_MAX : INT_MIN;
            }

            res = 10 * res + (str[i++] - '0');
        }

        return sign * res;
    }
};
```
