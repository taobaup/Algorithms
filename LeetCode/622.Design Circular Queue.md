class MyCircularQueue {
public:
    MyCircularQueue(int k) {
        data.resize(k);
        head = k - 1;
        tail = 0;
        size = k;
        cnt = 0;
    }
    
    bool enQueue(int value) {
        if (isFull())
        {
            return false;
        }

        // 不要写错 tail 的下标
        data[tail] = value;
        tail = (tail + 1) % size;
        ++cnt;

        return true;
    }
    
    bool deQueue() {
        // 不要写成 isFull
        if (isEmpty())
        {
            return false;
        }

        head = (head + 1) % size;
        --cnt;

        return true;
    }
    
    int Front() {
        // 不要遗漏 isEmpty() 的判断
        // 不要写成 return isEmpty() ? -1 : (head + 1) % size;
        return isEmpty() ? -1 : data[(head + 1) % size];
    }
    
    int Rear() {
        // 不要写成 return isEmpty() ? -1 : (tail - 1 + size) % size;
        return isEmpty() ? -1 : data[(tail - 1 + size) % size];
    }
    
    bool isEmpty() {
        // return cnt == 0 ? true : false;
        return cnt == 0;
    }
    
    bool isFull() {
        // return cnt == size ? true : false;
        return cnt == size;
    }

private:
    vector<int> data;
    int head, tail, size, cnt;
};
