class Solution {
public:
	int maxArea(vector<int>& height) {
		int res = INT_MIN;

		int start = 0;
		int end = height.size() - 1;

		while (start < end)
		{
			int h = min(height[start], height[end]);
			res = max(res, h * (end - start));

			if (height[start] < height[end])
			{
				++start;
			}
			else
			{
				--end;
			}
		}

		return res;
	}
};

class Solution {

    int maxArea(vector<int>& height) {
        int res = 0;
        int i = 0;
        int j = height.size() - 1;
        
        while (i < j) 
        {
            int h = min(height[i], height[j]);
            res = max(res, h * (j - i));

            while (i < j && height[i] == h) 
            {
            	++i;
            }

            while (i < j && height[j] == h)
            {
            	--j;
            }
        }

        return res;
    }
};
