```
// Definition for a binary tree node.
struct TreeNode {
    int val;
    TreeNode *left;
    TreeNode *right;
    TreeNode(int x) : val(x), left(NULL), right(NULL) {}
};
```


```
class Solution {
public:
    void flatten(TreeNode* root) {
        vector<TreeNode*> res;

        if(root == NULL)
            return;

        stack<TreeNode*> s;
        TreeNode *p = root;

        while(!s.empty() || p != NULL)
        {
            if(p != NULL)
            {
                s.push(p);
                res.push_back(p);
                p = p->left;
            }
            else
            {
                TreeNode *t = s.top();
                s.pop();
                p = t->right;
            }
        }

        root = res[0];
        for(int i = 1; i < res.size(); ++i)
        {
            root->left = NULL;
        	root->right = res[i];
            root = root->right;
        }
    }
};
```
