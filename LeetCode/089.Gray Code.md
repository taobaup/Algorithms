```
Int    Gray Code    Binary
 0  　　  000        000
 1  　　  001        001
 2   　 　011        010
 3   　 　010        011
 4   　 　110        100
 5   　 　111        101
 6   　 　101        110
 7   　　 100        111
```



```
class Solution {
public:
    vector<int> grayCode(int n) {
        vector<int> res;

        for(int i = 0; i < pow(2, n); ++i)
        {
            res.push_back((i >> 1) ^ i);
        }

        return res;
    }
};
```
